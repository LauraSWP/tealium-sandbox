Consent Banner (All Tags - Pre Loader)
// Show consent banner if no consent preference exists
if (!localStorage.getItem('consent_preference')) {
  var banner = document.createElement('div');
  banner.id = 'consent-banner';
  banner.innerHTML = 'We use cookies. ';
  banner.style.cssText = 'position:fixed;bottom:0;width:100%;background:#333;color:#fff;padding:10px;z-index:9999;';
  document.body.appendChild(banner);
  
  window.acceptConsent = function() {
    localStorage.setItem('consent_preference', 'accepted');
    document.getElementById('consent-banner').remove();
    utag.view(utag_data); // Reload tags with consent
  };
}
